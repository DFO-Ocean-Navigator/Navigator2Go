54 /*
55  * Class QEasyDownloader <- Inherits QObject
56  * --------------------
57  *
58  * The main class that handles downloads like a pro.
59  *
60  * Constructors:
61  * QEasyDownloader(QObject *parent = NULL , QNetworkAccessManager *toUseManager = NULL);
62  *
63  * Assigns a parent if the user gives one or assigns a QNetworkAccessManager if the user
64  * already uses a single NetworkManager for his/her application.
65  *
66  * Methods:
67  * void setDebug(bool) - Enable or Disable Debuging
68  * void setIterated(bool) - Enable iterated Downloading.
69  * ( i.e Download a file in Queue , Stop , Get Approved then Download again.)
70  * void setResumeDownloads(bool) - Enable or Disable Resuming of Downloads!
71  *
72  * Warning: Disabling Resuming of Downloads will overwrite the file if found!
73  *
74  * void setTimeoutTime(int) - sets the timeout time (in miliseconds) for a request! default is 5000 = 5 secs
75  * void setRetryTime(int) - sets the retry time (in miliseconds) for a request! default is 6000 = 6 secs
76  *
77  * Public Slots:
78  * void Download(const QString& , const QString&) - Download a file and save it in the location provided.
79  * void Download(const QString&) - Simply download a file.
80  * void Pause() - Pause the current download.
81  * void Resume() - Resume any paused download.
82  * bool HasNext() - If iterated is true then this function will return true if download queue is not empty.
83  * void Next() - Downloads the next entry in the download queue.
84  *
85  *
86  * Signals:
87  * void Finished() - Emitted when all jobs are done.
88  * void DownloadFinished(const QUrl &url, const QString& fileName) - Emitted when a single file is downloaded.
89  * void DownloadProgress(qint64 bytesReceived,
90  * qint64 bytesTotal,
91  * int percent,
92  * double speed,
93  * const QString &unit,
94  * const QUrl &url,
95  * const QString &fileName) - Full Download Progress , Emitted on every download.
96  * void Error(QNetworkReply::NetworkError errorCode,
97  * const QUrl &url,
98  * const QString &fileName) - Emitted on error.
99  * void Timeout(const QUrl &url, const QString &fileName) - Emitted when there is a timeout.
100  *
101  * void Paused(const QUrl &url , const QString &fileName) - Emitted when paused a download.
102  * void Resumed(const QUrl &url , const QString &fileName) - Emitted when resumed a download.
103  * void Debugger(const QString &msg) - Emitted when a debug message is passed.
104  *
105  *
106 */
